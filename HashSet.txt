class A{
    int id;

    {
        this.id=id;
        this.name=name;
        this.address=address;
        this.Email=Email;
    }
}
public class Main {
    public static void main(String[] args) {
     ArrayList<A> al=new ArrayList();

     A k=new A(1,"mohit","mohit@gmail","pratapnagar");
     A k1=new A(2,"rohit","rohit@gmail","gopalpura");
     A k2=new A(3,"sohit","sohit@gmail","murlipura");

     al.add(k);
     al.add(k1);
     al.add(k2);

     for(A i:al)
     {
         System.out.println(i.id+" "+i.name+" "+i.Email+" "+i.address);
     }
    }
}
